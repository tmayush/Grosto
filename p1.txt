1) Numpy
import numpy as np

# Create a NumPy array
my_array = np.array([1, 2, 3, 4, 5])

# Display the array
print(my_array)

# Perform basic operations on the array
mean = np.mean(my_array)
sum_elements = np.sum(my_array)
max_value = np.max(my_array)

print("\nMean:", mean)
print("Sum of Elements:", sum_elements)
print("Maximum Value:", max_value)

2) Pandas
import pandas as pd

# Create a pandas DataFrame
data = {'Name': ['Alice', 'Bob', 'Charlie'],
        'Age': [25, 30, 35]}
df = pd.DataFrame(data)
young_people = df[df['Age'] < 30]
print("\nYoung People:")
print(young_people)

3) Matlplotlib
import matplotlib.pyplot as plt

# Data for the plot
x = [1, 2, 3, 4, 5]
y = [10, 15, 13, 18, 20]

# Create a line plot
plt.plot(x, y)

# Add labels and a title
plt.xlabel('X-axis')
plt.ylabel('Y-axis')
plt.title('Simple Line Plot')

# Display the plot
plt.show()




import pymongo
# Establish a connection to MongoDB (default connection to localhost)
client = pymongo.MongoClient("mongodb://localhost:27017/")
# Create or use an existing database
db = client["mydatabase"]
# Create a collection or use an existing one
collection = db["mycollection"]
# Sample data to insert
sample_data = [
 {"name": "John", "age": 30, "city": "New York"},
 {"name": "Alice", "age": 25, "city": "San Francisco"},
 {"name": "Bob", "age": 35, "city": "Los Angeles"}
]
# Insert the sample data into the collection
collection.insert_many(sample_data)
# Retrieve and print all data in the collection
cursor = collection.find({})
for document in cursor:
 print(document)
# Close the MongoDB connection
client.close()
